#define MAX 100005
int tree[4*MAX];
int base;
void init(int n){
    base = (1<<(int)(ceil(log2(n+2)))) -1;
    for(int lx = 0;lx < 4*MAX;lx++)
        tree[lx] = 0;
    return;
}
void touch(int a){
    int prc = a + base + 2;
    tree[prc]++;
    for(prc>>=1;prc;prc>>=1)
        tree[prc] = tree[prc*2] + tree[prc*2 + 1];
    return;
}
int qry(int a, int b){
    int ret = 0;
    for(a += base + 1, b += base+3;
        a^b^1;a>>=1, b>>=1){
        if(~a&1) ret += tree[a^1];
        if(b&1) ret += tree[b^1];
    }
    return ret;
}

如此qry函數便可以知道 $[a, b]$ 之間有多少的元素，進而用二分搜實作出kth_element
