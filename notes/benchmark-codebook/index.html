struct MemoryInfo {
    // kb
    double rss;
    double sharedMemory;
    double privateMemory;
};

MemoryInfo getMemoryInfo() {
    int tSize = 0, resident = 0, share = 0;
    std::ifstream buffer("/proc/self/statm");
    buffer >> tSize >> resident >> share;
    buffer.close();

    long pageSizeKB = sysconf(_SC_PAGE_SIZE) / 1024;
    double rss = resident * pageSizeKB;
    double sharedMem = share * pageSizeKB;
    return MemoryInfo{
        .rss = rss,
        .sharedMemory = sharedMem,
        .privateMemory = rss - sharedMem,
    };
}

uint64_t getDiskUsage(const std::string& path) {
    uint64_t usedKB;
    std::ifstream buffer("du -s " + path);
    buffer >> usedKB;
    buffer.close();
    return usedKB;
}

class CSVFile {
public:
    CSVFile(const std::string& filename) {
        fp.open(filename);
        if (!fp) {
            throw std::runtime_error("CSVFile open error");
        }
    }

    void write(const std::vector<uint64_t>& vals) {
        for (size_t lx = 0; lx < vals.size(); ++lx) {
            fp << vals[lx];
            if (lx + 1 < vals.size()) {
                fp << ",";
            }
        }
        fp << std::endl;
    }

private:
    std::ofstream fp;
};

void writeValues(const std::string& filename,
                 const std::vector<uint64_t>& values) {
    std::ofstream fp;
    fp.open(filename);
    if (!fp) {
        throw std::runtime_error(filename + " open error");
    }

    for (const auto& v : values) {
        fp << v << std::endl;
    }
}

std::vector<int> shuffleIndex(int n) {
    std::vector<int> ret(n);
    for (int lx = 0; lx < n; lx++) {
        ret[lx] = lx;
    }
    
    // Just for testing, hence using random_shuffle
    std::random_shuffle(ret.begin(), ret.end());
    return ret;
}

// Split data into n parts, return k parts of n parts.
std::vector<std::vector<int>> split(const std::vector<int>& arr, int n, int k) {
    size_t sz = arr.size() / n;
    std::vector<std::vector<int>> ret(k, std::vector<int>(sz));

    for (int lx = 0; lx < k; lx++) {
        for (int ly = 0; ly < sz; ly++) {
            ret[lx][ly] = arr[(lx % n) * sz + ly];
        }
    }
    return ret;
}
